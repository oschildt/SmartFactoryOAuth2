 18: {
 19:     /**
 20:      * Initializes the authentication manager with parameters. 
 21:      *
 22:      * @param array $parameters

 24:      *
 25:      * @return boolean
 26:      * The method should return true upon successful initialization, otherwise false.

 29:      */

 31:     
 32:     /**
 33:      * Authenticates the user based on the specified credentials and writes the response.
 34:      *
 35:      * It should use {@see \OAuth2\Interfaces\IUserAuthenticator} for the authentication.
 36:      *
 37:      * @param array $credentials
 38:      * The credentials for the authentication. They may vary depending on implementation.
 39:      *
 40:      * @param array $response
 41:      * The response upon successful authentication. The expected properties are:
 42:      *
 43:      * - $response["user_id"] - id of the user.
 44:      *
 45:      * - $response["jwt_access_token"] - jwt access token generated upon successful authentication. JWT access
 46:      * token is the access token encoded and signed by the JWT standard approach.
 47:      *
 48:      * - $response["refresh_token"] - refresh token generated upon successful authentication.
 49:      *
 50:      * - $response["access_token_ttl"] - time to live in seconds for the access token.
 51:      *
 52:      * - $response["refresh_token_ttl"] - time to live in seconds for the refresh token.
 53:      *
 54:      * @return boolean
 55:      * The method should return true upon success, otherwise false.
 56:      *
 57:      * @throws \SmartFactory\SmartException
 58:      * It might throw an exception in the case of any errors.
 59:      *
 60:      * @author Oleg Schildt
 61:      */
 62:     public function authenticateUser($credentials, &$response);
 63:     
 64:     /**
 65:      * Refreshes the access token by using the valid refresh token, if the access token is expired.
 66:      *
 67:      * @param string $refresh_token
 68:      * The refresh token generated upon successful authentication.
 69:      *
 70:      * @param string $user_id
 71:      * The user id for which the refresh token was issued.
 72:      *
 73:      * @param string $client_id
 74:      * The client id for which the refresh token was issued.
 75:      *
 76:      * @param array $response
 77:      * The response upon successful refresh. The expected properties are:
 78:      *
 79:      * - $response["user_id"] - id of the user.
 80:      *
 81:      * - $response["jwt_access_token"] - jwt access token generated upon successful authentication. JWT access
 82:      * token is the access token encoded and signed by the JWT standard approach.
 83:      *
 84:      * - $response["refresh_token"] - refresh token generated upon successful refresh. The refresh token is also renewed!
 85:      *
 86:      * - $response["access_token_ttl"] - time to live in seconds for the access token.
 87:      *
 88:      * - $response["refresh_token_ttl"] - time to live in seconds for the refresh token.
 89:      *
 90:      * @return boolean
 91:      * The method should return true upon success, otherwise false.
 92:      *
 93:      * @throws \SmartFactory\SmartException
 94:      * It might throw an exception in the case of any errors.
 95:      *
 96:      * @author Oleg Schildt
 97:      */
 98:     public function refreshAccessToken($refresh_token, $user_id, $client_id, &$response);
 99:     
100:     /**

102:      *
103:      * @param string $jwt_access_token
104:      * The jwt access token generated upon successful authentication.
105:      *
106:      * @param boolean $check_on_server
107:      * This flag defines whether only the signatire should be checked or also the
108:      * authentication server should be asked.
109:      *
110:      * @return array|false
111:      * The method should return the payload array on successful verification, otherwise false.
112:      *
113:      * @throws \SmartFactory\SmartException
114:      * It might throw an exception in the case of any errors.
115:      *
116:      * @author Oleg Schildt
117:      */
118:     public function verifyJwtAccessToken($jwt_access_token, $check_on_server = true);
119:     
120:     /**
121:      * Invalidates all token records for the user.
122:      *
123:      * The invalidation by the user id should be asserted by a valid client id and
124:      * a valid refresh token. If the refresh token is expired, the invalidation
125:      * can be done only after authentication.
126:      *
127:      * The invalidation is done by using {@see \OAuth2\Interfaces\ITokenStorage::deleteTokenRecordByKey()} with the key 'user_id'.
128:      *
129:      * @param string $user_id
130:      * The user id for which the refresh token was issued.
131:      *
132:      * @param string $client_id
133:      * The client id for which the refresh token was issued.
134:      *
135:      * @param string $refresh_token
136:      * The refresh token generated upon successful authentication.
137:      *
138:      * @return boolean
139:      * The method should return true on successful invalidation, otherwise false.
140:      *
141:      * @throws \SmartFactory\SmartException
142:      * It might throw an exception in the case of any errors.
143:      *
144:      * @author Oleg Schildt
145:      */
146:     public function invalidateUser($user_id, $client_id, $refresh_token);
147:     
148:     /**
149:      * Invalidates all token records for the client.
150:      *
151:      * The invalidation by the client id should be asserted by a valid user id and
152:      * a valid refresh token. If the refresh token is expired, the invalidation
153:      * can be done only after authentication.
154:      *
155:      * The invalidation is done by using {@see \OAuth2\Interfaces\ITokenStorage::deleteTokenRecordByKey()} with the key 'client_id'.
156:      *
157:      * @param string $user_id
158:      * The user id for which the refresh token was issued.
159:      *
160:      * @param string $client_id
161:      * The client id for which the refresh token was issued.
162:      *
163:      * @param string $refresh_token
164:      * The refresh token generated upon successful authentication.
165:      *
166:      * @return boolean
167:      * The method should return true on successful invalidation, otherwise false.
168:      *
169:      * @throws \SmartFactory\SmartException
170:      * It might throw an exception in the case of any errors.
171:      *
172:      * @author Oleg Schildt
173:      */
174:     public function invalidateClient($user_id, $client_id, $refresh_token);
175:     
176:     /**
177:      * Invalidates the token record for the jwt access token.
178:      *
179:      * The invalidation is done by using {@see \OAuth2\Interfaces\ITokenStorage::deleteTokenRecordByKey()} with the key 'access_token'.
180:      *
181:      * The origin access token is extracted from the jwt payload.
182:      *
183:      * @param string $jwt_access_token
184:      * The jwt access token generated upon successful authentication.
185:      *
186:      * @return boolean
187:      * The method should return true on successful invalidation, otherwise false.
188:      *
189:      * @throws \SmartFactory\SmartException
190:      * It might throw an exception in the case of any errors.
191:      *
192:      * @author Oleg Schildt
193:      */
194:     public function invalidateJwtAccessToken($jwt_access_token);

198:      *

200:      *
201:      * @param string $refresh_token
202:      * The refresh token generated upon successful authentication.
203:      *
204:      * @return boolean

206:      *
207:      * @throws \SmartFactory\SmartException

209:      *
210:      * @author Oleg Schildt

212:     public function invalidateRefreshToken($refresh_token);
