 12: /**
 13:  * Interface for storing and validating of the stored tokens. It can be implemnted for a database or
 14:  * for a cache solution like Redis.
 15:  *
 16:  * @author Oleg Schildt
 17:  */
 18: interface ITokenStorage extends IInitable
 19: {
 20:     /**
 21:      * Saves the token record.
 22:      *
 23:      * It is reccomnended to look for an existing record by user_id and client_id, and if it exists,
 24:      * it should be overwritten.
 25:      *
 26:      * @param array $token_record
 27:      * The token record may vary depending on implementation. The expected parameters are:
 28:      *
 29:      * - $token_record["user_id"] - id of the user.
 30:      *
 31:      * - $token_record["client_id"] - id of the client (device token etc.).
 32:      *
 33:      * - $token_record["access_token"] - access token generated upon successful authentication.
 34:      *
 35:      * - $token_record["refresh_token"] - refresh token generated upon successful authentication.
 36:      *
 37:      * - $token_record["access_token_ttl"] - time to live in seconds for the access token.
 38:      *
 39:      * - $token_record["refresh_token_ttl"] - time to live in seconds for the refresh token.
 40:      *
 41:      * @return boolean
 42:      * The method should return true upon successful save operation, otherwise false.
 43:      *
 44:      * @throws \Exception
 45:      * It might throw an exception in the case of any system errors.
 46:      *
 47:      * @author Oleg Schildt
 48:      */
 49:     public function saveTokenRecord(&$token_record);
 50:     
 51:     /**
 52:      * Deletes the token record by a key.
 53:      *
 54:      * @param string $key
 55:      * The name of the key, by which the record should be located and deleted. The possible keys are:
 56:      *
 57:      * - 'user_id' - id of the user.
 58:      *
 59:      * - 'client_id' - id of the client (device token etc.).
 60:      *
 61:      * - 'access_token' - access token generated upon successful authentication.
 62:      *
 63:      * - 'refresh_token' - refresh token generated upon successful authentication.
 64:      *
 65:      * @param string $value
 66:      * The vale of the key, by which the record should be located and deleted.
 67:      *
 68:      * @return boolean
 69:      * The method should return true upon successful deletion, otherwise false.
 70:      *
 71:      * @throws \Exception
 72:      * It might throw an exception in the case of any system errors.
 73:      *
 74:      * @used_by IOAuthManager::invalidateUser()
 75:      * @used_by IOAuthManager::invalidateClient()
 76:      * @used_by IOAuthManager::invalidateJwtAccessToken()
 77:      * @used_by IOAuthManager::invalidateRefreshToken()
 78:      *
 79:      * @author Oleg Schildt
 80:      */
 81:     public function deleteTokenRecordByKey($key, $value);
 82:     
 83:     /**
 84:      * Verifies the access token.
 85:      *
 86:      * @param string $access_token
 87:      * The access token to be verified.
 88:      *
 89:      * @param string $user_id
 90:      * The user id for which the access token was issued.
 91:      *
 92:      * @param string $client_id
 93:      * The client id for which the access token was issued.
 94:      *
 95:      * @return boolean
 96:      * The method should return true upon successful verification, otherwise false.
 97:      *
 98:      * @throws \Exception
 99:      * It might throw an exception in the case of any errors.
100:      *
101:      * @throws \OAuth2\InvalidTokenException
102:      * It should throw the InvalidTokenException if the access token is invalid.
103:      *
104:      * @throws \OAuth2\MissingParametersException
105:      * It should throw the MissingParametersException if any required paramters are empty.
106:      *
107:      * @author Oleg Schildt
108:      */
109:     public function verifyAccessToken($access_token, $user_id, $client_id);
110:     
111:     /**
112:      * Verifies the refresh token.
113:      *
114:      * @param string $refresh_token
115:      * The refresh token to be verified.
116:      *
117:      * @param string $user_id
118:      * The user id for which the refresh token was issued.
119:      *
120:      * @param string $client_id
121:      * The client id for which the refresh token was issued.
122:      *
123:      * @return boolean
124:      * The method should return true upon successful verification, otherwise false.
125:      *
126:      * @throws \Exception
127:      * It might throw an exception in the case of any system errors.
128:      *
129:      * @throws \OAuth2\InvalidTokenException
130:      * It should throw the InvalidTokenException if the refresh token is invalid.
131:      *
132:      * @throws \OAuth2\MissingParametersException
133:      * It should throw the MissingParametersException if any required paramters are empty.
134:      *
135:      * @author Oleg Schildt
136:      */
137:     public function verifyRefreshToken($refresh_token, $user_id, $client_id);
138: } // ITokenStorage
